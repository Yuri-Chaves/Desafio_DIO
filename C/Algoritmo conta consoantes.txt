#include <stdio.h>
#include <string.h>

int conta( char nome[])
{
	
	int i, tam, qtd;
	
	tam = strlen(nome);
	nome = strupr (nome);
	qtd = 0;
	for (i = 0; i < tam; i++){
		if((nome[i] != 'A') && (nome[i] != 'E') && (nome[i] != 'I') && (nome[i] != 'O') && (nome[i] != 'U')){
			qtd ++;
		}
	}
	return qtd;
}

int main()
{

	int total;
	char palavra[30];
	printf("Informe a string:");

	printf("***********************");
